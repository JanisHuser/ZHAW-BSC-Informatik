	\subsection{Subqueries (in WHERE) mit EXISTS}
	

\begin{minted}{sql}
SELECT *
FROM Besucher AS x
WHERE EXISTS (SELECT 1
	FROM Besucher AS y
	WHERE x.Vorname=y.Vorname AND
	NOT (x.Name=y.Name));
\end{minted}
